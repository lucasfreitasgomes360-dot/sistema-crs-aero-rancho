<script src="https://www.gstatic.com/firebasejs/9.6.1/firebase-app-compat.js"></script>
<script src="https://www.gstatic.com/firebasejs/9.6.1/firebase-database-compat.js"></script>
<script>
    const firebaseConfig = { /* MESMA CONFIG DO OUTRO */ };
    firebase.initializeApp(firebaseConfig);
    const db = firebase.database();

    function enviarChamada(nomeRepetido = null, generoRepetido = null) {
        const nome = nomeRepetido || document.getElementById('nome').value;
        const local = document.getElementById('setor').value;
        const genero = generoRepetido || document.getElementById('genero').value;

        if(!nome) return alert("Digite o nome!");

        const dados = { nome, local, genero, timestamp: Date.now() };
        
        // Envia para a TV
        db.ref('chamada_atual').set(dados);
        
        // Salva no histórico do setor para o "Chamar Novamente"
        const idSetor = local.replace(/\s+/g, '_').toLowerCase();
        db.ref(`historico/${idSetor}`).push({ ...dados, hora: new Date().toLocaleTimeString() });
        
        document.getElementById('nome').value = "";
    }

    // Carregar histórico do setor selecionado
    document.getElementById('setor').onchange = carregarHistorico;
    function carregarHistorico() {
        const local = document.getElementById('setor').value;
        const idSetor = local.replace(/\s+/g, '_').toLowerCase();
        
        db.ref(`historico/${idSetor}`).limitToLast(5).on('value', snap => {
            let html = "";
            snap.forEach(item => {
                const c = item.val();
                html += `<div>${c.hora} - ${c.nome} <button class="re-chamar" onclick="enviarChamada('${c.nome}','${c.genero}')">Re-chamar</button></div>`;
            });
            document.getElementById('lista-historico').innerHTML = html;
        });
    }
    carregarHistorico();
</script>
